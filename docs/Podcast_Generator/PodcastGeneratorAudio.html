Podcast_Generator.PodcastGeneratorAudio

PodcastGeneratorAudio.py

Module responsable de la génération vocale à partir des scripts de podcast.

Rôle :

  • Convertir chaque réplique en audio à l’aide de XTTS (via TTSWrapper)
  • Appliquer les paramètres de tonalité (vitesse, émotion) issus des presets
  • Associer les voix aux personnages, lire le titre, et sauvegarder les fichiers
  • Gérer la fusion finale des fichiers audio en un seul mix .wav

Structure :

  • Fonctions internes (résolution de voix, découpage, etc.)
  • Fonctions de génération (sentence, discussion complète)
  • Outils audio (play, listage, fusion)

 59def sanitize_filename(text):
 60    """
 61    Nettoie une chaîne de caractères pour l'utiliser comme nom de fichier.
 62
 63    Args:
 64        text (str): Chaîne à nettoyer.
 65
 66    Returns:
 67        str: Chaîne nettoyée sans caractères interdits.
 68
 69    Remarque:
 70        Supprime les caractères incompatibles avec les systèmes de fichiers.
 71    """
 72    invalid_chars = '<>:"/\\|?*'
 73    return ''.join(c for c in text if c not in invalid_chars)

 76def _resolve_voice(voice: str, voice_dir: str = None) -> dict:
 77    """
 78    Résout dynamiquement une voix pour XTTS à partir de :
 79    - soit un nom de voix intégré au modèle XTTS (ex: "fr_1")
 80    - soit un fichier personnalisé .wav situé dans le dossier `voice_dir` (par défaut "Voices/")
 81
 82    Args:
 83        voice (str): Nom logique (voix XTTS, ex: "fr_1") OU nom de fichier voix (ex: "Hanekawa.wav").
 84        voice_dir (str, optional): Dossier contenant les voix personnalisées (par défaut "Voices/").
 85
 86    Returns:
 87        dict: Paramètre à passer dans XTTS :
 88              - {"speaker": <nom>} pour les voix intégrées
 89              - {"speaker_wav": <chemin>} pour un fichier personnalisé
 90
 91    Raises:
 92        FileNotFoundError: Si un fichier .wav est précisé mais introuvable.
 93        ValueError: Si un nom XTTS inconnu est utilisé.
 94
 95    Exemple:
 96        _resolve_voice("fr_1")
 97        >> {"speaker": "fr_1"}
 98
 99        _resolve_voice("Hanekawa.wav")
100        >> {"speaker_wav": "Voices/Hanekawa.wav"}
101
102    Remarques:
103        - Si l'argument se termine par .wav, on cherche un fichier.
104        - Sinon, on vérifie s'il fait partie des voix intégrées du modèle.
105        - La recherche du fichier est relative à `voice_dir`, ou au dossier par défaut "Voices/".
106    """
107    if voice_dir is None:
108        voice_dir = VOICE_DIR
109
110    voice_path = os.path.join(voice_dir, voice)
111
112    if voice.lower().endswith(".wav"):
113        if os.path.isfile(voice_path):
114            return {"speaker_wav": voice_path}
115        else:
116            raise FileNotFoundError(f"Fichier voix introuvable : {voice_path}")
117
118    available = get_available_speakers()
119    if voice not in available:
120        raise ValueError(f"Voix '{voice}' inconnue. Voix disponibles : {list(available)}")
121    return {"speaker": voice}

124def _split_text(text: str, max_len: int = MAX_LENGTH):
125    """
126    Découpe un texte trop long en segments courts compatibles XTTS.
127
128    Args:
129        text (str): Texte brut à découper.
130        max_len (int): Longueur maximale par segment (limite XTTS).
131
132    Returns:
133        list[str]: Liste de segments courts prêts pour la synthèse.
134
135    Remarque:
136        Priorise la séparation par phrases, sinon coupe par mots.
137        Essentiel, car XTTS couperait sinon de manière arbitraire.
138    """
139    sentences = re.split(r'(?<=[.!?])\s+', text.strip())
140    chunks = []
141    for sentence in sentences:
142        if len(sentence) <= max_len:
143            chunks.append(sentence.strip())
144        else:
145            # Split par mots si la phrase est trop longue
146            words = sentence.split()
147            current = ""
148            for word in words:
149                if len(current) + len(word) + 1 <= max_len:
150                    current += (" " + word) if current else word
151                else:
152                    chunks.append(current.strip())
153                    current = word
154            if current:
155                chunks.append(current.strip())
156    return chunks

158def play_audio(file_path: str):
159    """
160    Lit un fichier audio localement (.wav) et affiche des messages de suivi.
161
162    Args:
163        file_path (str): Chemin vers le fichier à lire.
164
165    Returns:
166        None
167
168    Raises:
169        FileNotFoundError: Si le fichier n'existe pas.
170        Exception: En cas d'erreur de lecture.
171
172    Notes:
173        - Affiche un message lors de la lecture et en cas d'erreur.
174    """
175    try:
176        if not os.path.exists(file_path):
177            raise FileNotFoundError(f"Le fichier n'existe pas : {file_path}")
178        sound = AudioSegment.from_file(file_path)
179        print(f"Lecture audio en cours : {file_path}")
180        play(sound)
181    except Exception as e:
182        print(f"[ERREUR] Impossible de lire le fichier : {file_path}")
183        print(f"[Exception] {e}")

185def convert_wav_to_mp3(wav_path: str) -> str:
186    """
187    Convertit un fichier .wav en .mp3 dans le même dossier.
188
189    Args:
190        wav_path (str): Chemin vers le fichier .wav source.
191
192    Returns:
193        str: Chemin du fichier .mp3 généré.
194
195    Raises:
196        FileNotFoundError: Si le fichier d'entrée n'existe pas.
197        ValueError: Si l'extension n'est pas .wav
198    """
199    source = Path(wav_path)
200    if not source.exists():
201        raise FileNotFoundError(f"Fichier introuvable : {wav_path}")
202    if source.suffix.lower() != ".wav":
203        raise ValueError("Le fichier doit être un .wav")
204
205    mp3_path = source.with_suffix(".mp3")
206
207    audio = AudioSegment.from_wav(source)
208    audio.export(mp3_path, format="mp3")
209    return str(mp3_path)

309def create_sentence(
310    voice: str,
311    ton: str,
312    texte: str,
313    language: str = "fr",
314    output_path: str = "output",
315    filename: str = "sentence.wav"
316) -> str:
317    """
318    Génère un fichier audio (.wav) contenant une phrase vocalisée avec XTTS.
319
320    Args:
321        voice (str): Nom de la voix intégrée (ex: "fr_1") ou nom de fichier .wav personnalisé.
322        ton (str): Nom du ton à utiliser (doit exister dans les presets sinon "neutre").
323        texte (str): Texte à synthétiser.
324        language (str, optional): Langue compatible XTTS. Par défaut : "fr".
325            Langues supportées :
326            - "fr" : français
327            - "en" : anglais
328            - "ja" : japonais
329            - "zh-cn" : chinois simplifié
330            - "zh-tw" : chinois traditionnel (Taïwan)
331        output_path (str, optional): Dossier de sortie où sauvegarder l’audio.
332        filename (str, optional): Nom du fichier à générer.
333
334    Returns:
335        str: Chemin complet vers le fichier audio généré.
336
337    Remarques:
338        - Si le texte dépasse la limite XTTS (273 caractères), il sera découpé automatiquement.
339        - Tous les morceaux sont ensuite fusionnés en un seul fichier.
340
341    Exemple:
342        create_sentence("fr_1", "calme", "Bonjour, ceci est un test.")
343    """
344    os.makedirs(output_path, exist_ok=True)
345    final_path = os.path.join(output_path, filename)
346
347    # Utilise le preset du ton si reconnu, sinon neutre
348    params = tone_presets.get(ton.lower(), {"speed": 1.0, "emotion": "neutral"})
349
350    voice_resolved = _resolve_voice(voice)
351    if "speaker_wav" in voice_resolved:
352        voice_resolved.pop("speaker", None)
353    else:
354        voice_resolved.pop("speaker_wav", None)
355
356    if len(texte) <= MAX_LENGTH:
357        tts.tts_to_file(
358            text=texte,
359            file_path=final_path,
360            language=language,
361            emotion=params["emotion"],
362            speed=params["speed"],
363            **voice_resolved
364        )
365        return final_path
366
367    # Si le texte est trop long → découpage et fusion
368    parts = _split_text(texte)
369    all_audio = AudioSegment.empty()
370
371    for i, part in enumerate(parts):
372        part_path = os.path.join(output_path, f"_part_{i}.wav")
373        tts.tts_to_file(
374            text=part,
375            file_path=part_path,
376            language=language,
377            emotion=params["emotion"],
378            speed=params["speed"],
379            **voice_resolved
380        )
381        all_audio += AudioSegment.from_wav(part_path)
382        os.remove(part_path)
383
384    all_audio.export(final_path, format="wav")
385    return final_path

387def create_discussion(meta: dict, participants: dict, dialogues: list[dict], presentateur: str = None, output_base: str = None):
388    """
389    Génère tous les fichiers audio pour un script de podcast (titre + dialogues).
390
391    Args:
392        meta (dict): Métadonnées du podcast, avec au minimum 'titre', 'date' et 'lang'.
393        participants (dict): Dictionnaire {nom_personnage: nom_voix}.
394        dialogues (list[dict]): Liste de blocs avec : 'name', 'tone', 'index', 'text'.
395        presentateur (str, optional): Voix utilisée pour lire le titre. Si None, voix aléatoire.
396        output_base (str, optional): Dossier de destination. Si None, utilise le dossier du fichier source.
397
398    Returns:
399        str: Chemin absolu du dossier généré.
400    """
401    titre = meta.get("titre", "").strip() or "Podcast sans titre"
402    date = meta.get("date", dt.today().date().isoformat())
403    langue = meta.get("lang", "fr")
404
405    titre_sanitized = sanitize_filename(titre).replace(" ", "_")
406    heure_str = dt.now().strftime("%H%M%S")
407    dossier_nom = f"{titre_sanitized}__{date}__{heure_str}"
408
409    # Création du dossier cible (dans output_base si fourni, sinon à côté du fichier source)
410    base_path = Path(output_base).resolve() if output_base else Path().resolve()
411    folder_path = base_path / dossier_nom
412    folder_path.mkdir(parents=True, exist_ok=True)
413
414    # Choix du présentateur
415    all_voices = list_all_available_voices()
416    if not presentateur or presentateur not in all_voices:
417        if not all_voices:
418            raise ValueError("Aucune voix disponible pour le présentateur.")
419        presentateur = random.choice(all_voices)
420        print(f"Présentateur choisi aléatoirement : {presentateur}")
421
422    # Intro : le titre
423    titre_audio_path = folder_path / f"00__PRESENTATEUR__neutre__{date}.wav"
424    print(f"Lecture du titre : « {titre} »")
425    intro_path = create_sentence(presentateur, "neutre", titre, language=langue)
426    os.rename(intro_path, titre_audio_path)
427
428    max_index = len(dialogues)
429    # Génération des répliques
430    for bloc in dialogues:
431        nom = bloc["name"]
432        texte = bloc["text"]
433        ton = bloc["tone"]
434        index = bloc["index"]
435
436        ton_final = ton if ton in tone_presets else "neutre"
437        if nom not in participants:
438            raise ValueError(f"[❌] Voix manquante pour le personnage : {nom}")
439        voix = participants[nom]
440
441        print(f"Génération [{index}/{max_index}:{nom}:{ton_final}]")
442
443        nom_fichier = f"{str(index).zfill(2)}__{nom}__{ton_final}__{date}.wav"
444        chemin_fichier = folder_path / nom_fichier
445
446        audio_path = create_sentence(voix, ton_final, texte, language=langue)
447        os.rename(audio_path, chemin_fichier)
448
449    print(f"[✅] Discussion complète générée dans le dossier : {folder_path}")
450    return str(folder_path)

454def muxgenerateddiscussion(files: list[str], output_path: str, output_name: str = None) -> str:
455    """
456    Assemble les fichiers audio .wav dans l'ordre donné (alphabétique pré-trié) en un seul fichier .wav
457
458    Args:
459        files (list[str]): Liste des chemins de fichiers .wav à assembler
460        output_path (str): Dossier contenant les fichiers à assembler (le mix final ira dans un sous-dossier 'Mix')
461        output_name (str, optional): Nom du fichier final. Si None, utilise 'FINAL - <nom dossier>.wav'.
462
463   Returns:
464        str: Chemin absolu du fichier final généré.
465
466    Raises:
467        FileNotFoundError: Si un fichier listé est introuvable.
468
469    Exemple:
470        muxgenerateddiscussion(["01.wav", "02.wav"], "Episode_1")
471    """
472    if not files:
473        raise ValueError("Aucun fichier à assembler fourni.")
474
475        # Création du sous-dossier 'Mix' dans le dossier d'entrée
476    mix_dir = os.path.join(output_path, "Mix")
477    os.makedirs(mix_dir, exist_ok=True)
478
479    # Nom de sortie par défaut : 'FINAL - <nom du dossier>.wav'
480    if output_name is None:
481        base_name = os.path.basename(os.path.normpath(output_path))
482        output_name = f"FINAL - {base_name}.wav"
483
484    # Chargement et assemblage
485    combined = AudioSegment.silent(duration=0)
486    for file in sorted(files):
487        if not os.path.exists(file):
488            raise FileNotFoundError(f"Fichier manquant : {file}")
489        segment = AudioSegment.from_wav(file)
490        combined += segment
491
492    output_file = os.path.join(mix_dir, output_name)
493    combined.export(output_file, format="wav")
494
495    return output_file

498def GenerateAndMux(full_dialogue_path: str, participants: dict = None, presentateur: str = None, output_base: str = None) -> str:
499    """
500    Génère un podcast audio complet depuis un fichier dialogue, puis assemble tous les fichiers audio.
501
502    Args:
503        full_dialogue_path (str): Fichier .txt contenant les dialogues formatés.
504        participants (dict, optional): Dictionnaire {personnage: voix}. Si None, détection automatique.
505        presentateur (str, optional): Voix du titre (présentateur). Si None, aléatoire.
506        output_base (str, optional): Dossier de destination pour les fichiers générés.
507                                     Si None, on utilise le dossier de full_dialogue_path.
508
509    Returns:
510        str: Chemin du fichier audio final mixé (.wav).
511    """
512    # Timers Start
513    start_time = time.time()
514    print(f"Début GenerateAndMux: {datetime.now().strftime('%Y-%m-%d %H:%M')}")
515
516    with open(full_dialogue_path, "r", encoding="utf-8") as f:
517        content = f.read()
518    meta, dialogues = parse_dialogue_file(content)
519
520    if participants is None:
521        names = extract_character_names_from_dialogue_file(full_dialogue_path)
522        participants = assign_voices_by_gender(names)
523
524    # Si output_base est None → on prend le dossier du fichier de dialogue
525    if output_base is None:
526        output_base = str(Path(full_dialogue_path).resolve().parent)
527
528    folder_path = create_discussion(meta, participants, dialogues, presentateur, output_base)
529    folder_path = Path(folder_path)
530
531    wavs = sorted(str(file) for file in folder_path.glob("*.wav"))
532    if not wavs:
533        raise ValueError("Aucun fichier à assembler fourni.")
534
535    final_mix_path = muxgenerateddiscussion(
536        files=wavs,
537        output_path=str(folder_path)
538    )
539
540    # Timers End
541    end_time = time.time()
542    elapsed = end_time - start_time
543    print(f"Fin GenerateAndMux: {datetime.now().strftime('%Y-%m-%d %H:%M')} — Temps écoulé : {elapsed:.2f} secondes")
544
545    return final_mix_path
