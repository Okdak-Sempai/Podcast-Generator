358    def import_file(self):
359        """Ouvre un dialogue pour importer un fichier et l'ajoute  la zone de texte."""
360        file_dialog = QFileDialog(self)
361        file_types = (
362            "Fichiers pris en charge (*.txt *.pdf *.docx *.tex *.md *.png *.jpg *.jpeg *.bmp);;"
363            "Textes (*.txt);;"
364            "PDF (*.pdf);;"
365            "Word (*.docx);;"
366            "LaTeX (*.tex);;"
367            "Markdown (*.md);;"
368            "Images (*.png *.jpg *.jpeg *.bmp)"
369        )
370        file_path, _ = file_dialog.getOpenFileName(self, "Ouvrir un fichier", "", file_types)
371
372        if file_path:
373            try:
374                texte = extract_file_handler(file_path)
375                self.text_area.setText(texte)
376            except Exception as e:
377                self.text_area.setPlainText(f"Erreur lors de l'importation : {e}")

379    def process_url(self):
380        """Rcupre l'URL entre et traite le texte."""
381        url = self.url_input.text()
382        if url:
383            try:
384                texte = extract_file_handler(url)
385                self.text_area.setText(texte)
386            except Exception as e:
387                self.text_area.setText(f"Erreur lors du traitement de l'URL : {str(e)}")
388        else:
389            self.text_area.setText("Veuillez entrer une URL valide.")

439    def generate_script(self):
440        texte_source = self.text_area.toPlainText().strip()
441        if not texte_source:
442            QMessageBox.warning(self, "Texte manquant", "Veuillez importer ou extraire du texte d'abord.")
443            return
444
445        try:
446            # Exemple d'appel  ta logique de parsing de dialogue
447            meta, dialogues = parse_dialogue_file(texte_source)
448
449            # Transformation du rsultat en texte pour l'afficher dans l'autre zone de texte
450            script_genere = "Titre: " + meta.get("titre", "Inconnu") + "\n\n"
451            for dialogue in dialogues:
452                script_genere += f"[{dialogue['index']}] {dialogue['name']} ({dialogue['tone']}): {dialogue['text']}\n"
453
454            # Affichage du script gnr
455            self.text_area_2.setPlainText(script_genere)
456
457        except Exception as e:
458            QMessageBox.warning(self, "Erreur de gnration", f"Erreur lors de la gnration du script: {str(e)}")

440    def generate_script(self):
441        texte_source = self.text_area.toPlainText().strip()
442        if not texte_source:
443            QMessageBox.warning(self, "Texte manquant", "Veuillez importer ou extraire du texte d'abord.")
444            return
445
446        try:
447            # Exemple d'appel  ta logique de parsing de dialogue
448            meta, dialogues = parse_dialogue_file(texte_source)
449
450            # Transformation du rsultat en texte pour l'afficher dans l'autre zone de texte
451            script_genere = "Titre: " + meta.get("titre", "Inconnu") + "\n\n"
452            for dialogue in dialogues:
453                script_genere += f"[{dialogue['index']}] {dialogue['name']} ({dialogue['tone']}): {dialogue['text']}\n"
454
455            # Affichage du script gnr
456            self.text_area_2.setPlainText(script_genere)
457
458        except Exception as e:
459            QMessageBox.warning(self, "Erreur de gnration", f"Erreur lors de la gnration du script: {str(e)}")
